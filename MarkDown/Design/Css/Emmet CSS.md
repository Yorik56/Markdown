# Propriétés

> border-bottom = bdb
> border-top = bdt
> font-size = fz
> margin-top = mt
> display : flex = df
> justify-content : center = jc:c
> align-items: center = ai:c
> fxd:c = flex-direction:column; 

# Valeurs

> ft18 = 
`font-size : 18 px;`
>  fz1vw = 
`font-size: 1vw;`

>    px → par défaut
>    p → %
>   e → em
>   r → rem
>   x → ex

> ft18r = 
`font-size : 18rem;`

# Unités multiples

> m10-20-30-40 = 
`margin : 10 20 30 40;` 

# Couleurs

> #1 → #111
> #e0 → #e0e0e0
> #fc0 → #ffcc00

> c#2 =
 `color : #222;`

# Propriétés multiples

> mt10+pt30+mb50+ml10 = 

```
margin-top : 10px;
padding-top : 30px;
margin-bottom : 50px;
margin-left : 10px;
```

> bd2-solid-#3+mb10+p20+c#e = 

```
border: 2px solid #333333;
margin-bottom: 10px;
padding: 20px;
color: #eeeeee;
```
